#include<iostream>
#include<cstdlib>
#include<string>
using namespace std;
int isOlder(string p1,string p2){
	


	
	int d1=p1[1]-'0'+(10*(p1[0]-'0'));	
	int d2=p2[1]-'0'+(10*(p2[0]-'0'));	//09-11-2000
	int m1=p1[4]-'0'+(10*(p1[3]-'0'));	//0123456789
	int m2=p2[4]-'0'+(10*(p2[3]-'0'));

	int y1=(p1[9]-'0')+(10*(p1[8]-'0'))+(100*(p1[7]-'0'))+(1000*(p1[6]-'0'));
	int y2=(p2[9]-'0')+(10*(p2[8]-'0'))+(100*(p2[7]-'0'))+(1000*(p2[6]-'0'));
	
	//It may not check error details like -some months has 31/30
	//But checks for month>12  day>31
	
	if((y1>y2)&&(y1>0))
		return 1;
	
	else if((y2>y1)&&(y2>0))
		return 2;
	
	else if((y1==y2)&&(y1&&y2>0)){
		
		if((m1>m2)&&((m1&&m2>0)&&(m1&&m2<13)))
			return 1;

		else if((m2>m1)&&((m1&&m2>0)&&(m1&&m2<13)))
			return 2;
		
		else if((m1==m2)&&((m1&&m2)>0&&(m1&&m2)<13)){
			
			if((d1>d2)&&(d1&&d2<31)&&(d1&&d2>0))
			return 1;
			else if((d2>d1)&&(d1&&d2<31)&&(d1&&d2>0))
			return 2;
			else 
			return 0;	
		}	
	}	
}

int main()
{
	string p1,p2;
	cin>>p1;
	cin>>p2;
	cout<<isOlder(p1,p2);
	return 0;
}
