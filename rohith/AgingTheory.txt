bool validDate(int d,int m,int y){
	
	if(m==1||m==3||m==5||m==7||m==8||m==10||m==12)
	{
		if(d==31)
			return true;
		else
			return false;	
	}
	else if(m==4||m==6||m==9||m==11)
	{
		if(d==30)
			return true;
		else
			return false;	
	}
	if(y%4==0&&m==2)
	{
		if(d==29)
			return true;
		else
			return false;
	}
	else if(y%4!=0&&m==2){
		if(d==28)
			return true;
		else
			return false;
	}
}

int isOlder(string p1,string p2){
	
	int d1=p1[1]-'0'+(10*(p1[0]-'0'));	
	int d2=p2[1]-'0'+(10*(p2[0]-'0'));	//09-11-2000
	int m1=p1[4]-'0'+(10*(p1[3]-'0'));	//0123456789
	int m2=p2[4]-'0'+(10*(p2[3]-'0'));
	int y1=(p1[9]-'0')+(10*(p1[8]-'0'))+(100*(p1[7]-'0'))+(1000*(p1[6]-'0'));
	int y2=(p2[9]-'0')+(10*(p2[8]-'0'))+(100*(p2[7]-'0'))+(1000*(p2[6]-'0'));
	
if(validDate(d1,m1,y1)&&validDate(d2,m2,y2))
{
	if((((y1&&y2>0)&&((m1&&m2>0)&&(m1&&m2<=12)))&&((d1&&d2>0)&&(d1&&d2<=31))))
	{
		if(y1>y2)
		return 1;
			
		else if(y2>y1)
		return 2;
		
		else if(y1==y2)
		{
			if(m1>m2)
				return 1;

			else if(m2>m1)
				return 2;
		
				else if(m1==m2)
				{
					if(d1>d2)
						return 1;
					else if(d2>d1)
						return 2;
					else
						return 0;	
				}	
		}	
	}
}
else 
return -1;	
	
}
